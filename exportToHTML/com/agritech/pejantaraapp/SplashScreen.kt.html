30   @Composable
31   fun SplashScreen(navController: NavHostController) {
32       val context = LocalContext.current
33       var visible by remember { mutableStateOf(false) }
34   
35       val alpha by animateFloatAsState(
36           targetValue = if (visible) 1f else 0f,
37           animationSpec = tween(durationMillis = 1000)
38       )
39   
40       LaunchedEffect(Unit) {
41           visible = true
42           delay(3000)
43           val user = FirebaseAuth.getInstance().currentUser
44           Log.d("SplashScreen", "User: $user, Verified: ${user?.isEmailVerified}")
45           if (user != null && user.isEmailVerified) {
46               navController.navigate("home") {
47                   popUpTo("splash") { inclusive = true }
48               }
49           } else {
50               navController.navigate("login") {
51                   popUpTo("splash") { inclusive = true }
52               }
53           }
54       }
55   
56       Box(
57           modifier = Modifier
58               .fillMaxSize()
59               .background(Color.White)
60               .graphicsLayer(alpha = alpha),
61           contentAlignment = Alignment.Center
62       ) {
63           Column(horizontalAlignment = Alignment.CenterHorizontally) {
64               Image(
65                   painter = painterResource(id = R.drawable.logo),
66                   contentDescription = "Logo Aplikasi",
67                   modifier = Modifier.size(150.dp)
68               )
69           }
70       }
71   }
