26       fun predictTrash(image: File): Flow<Resource<Long>> = flow {
27           try {
28               emit(Resource.Loading)
29   
30               // Validasi file gambar
31               val imageFile = image.reduceFileImage()
32               if (!imageFile.exists() || imageFile.length() == 0L) {
33                   emit(Resource.Error("Gambar tidak valid atau kosong."))
34                   return@flow
35               }
36   
37               // Siapkan request
38               val requestImageFile = imageFile.asRequestBody("image/jpeg".toMediaType())
39               val multipartBody = MultipartBody.Part.createFormData(
40                   "file",
41                   imageFile.name,
42                   requestImageFile
43               )
44   
45               // Prediksi di server
46               Log.d("TrashRepository", "Mengunggah gambar ke server...")
47               val response = apiService.predictTrash(multipartBody)
48   
49               // Simpan hasil prediksi ke database
50               val trash = Trash(
51                   predictedClass = response.predictedClass.data.nama,
52                   videoUrl = response.predictedClass.data.video,
53                   articleUrl = response.predictedClass.data.artikel,
54                   imageData = imageFile.readBytes(),
55               )
56               val trashId = trashDao.insert(trash)
57   
58               // Hapus data sampah lama
59               cleanUpOldTrash()
60               emit(Resource.Success(trashId))
61   
62               Log.i("TrashRepository", "Prediksi berhasil: ${response.predictedClass.data.nama}")
63           } catch (e: UnknownHostException) {
64               Log.e("TrashRepository", "Tidak dapat terhubung ke server: ${e.localizedMessage}")
65               emit(Resource.Error("Tidak dapat terhubung ke server. Periksa koneksi internet Anda."))
66           } catch (e: IOException) {
67               Log.e("TrashRepository", "Kesalahan jaringan: ${e.localizedMessage}")
68               emit(Resource.Error("Kesalahan jaringan. Coba lagi nanti."))
69           } catch (e: Exception) {
70               Log.e("TrashRepository", "Kesalahan tidak diketahui: ${e.localizedMessage}")
71               emit(Resource.Error("Kesalahan tidak diketahui: ${e.localizedMessage}"))
72           }
73       }
74   
75       fun fetchResultWithVideos(image: File): Flow<Resource<ResultUiState>> = flow {
76           try {
77               emit(Resource.Loading)
78   
79               // Langkah 1: Prediksi Sampah
80               val imageFile = image.reduceFileImage()
81               val requestImageFile = imageFile.asRequestBody("image/jpeg".toMediaType())
82               val multipartBody = MultipartBody.Part.createFormData(
83                   "file",
84                   imageFile.name,
85                   requestImageFile
86               )
87               val response = apiService.predictTrash(multipartBody)
88   
89               val predictedClass = response.predictedClass.data
90               if (predictedClass == null) {
91                   throw Exception("Data prediksi tidak valid.")
92               }
93   
94               val videos = tutorialRepository.getVideosByTrashType(predictedClass.nama)
95   
96               val resultUiState = ResultUiState(
97                   trashType = predictedClass.nama,
98                   imageUrl = predictedClass.video,
99                   videos = videos
100              )
101              emit(Resource.Success(resultUiState))
102  
103          } catch (e: Exception) {
104              Log.e("TrashRepository", "Kesalahan: ${e.localizedMessage}", e)
105              emit(Resource.Error(e.localizedMessage ?: "Kesalahan tidak diketahui."))
106          }
107      }

110      private suspend fun cleanUpOldTrash() {
111          val trashList = trashDao.getAllSync()
112          if (trashList.size > 30) {
113              val trashToDelete = trashList.take(trashList.size - 30)
114              trashDao.deleteTrashList(trashToDelete)
115          }
116      }
117  
118      fun getTrashHistory(): LiveData<List<Trash>> = trashDao.getAll()
119  
120      fun getTrash(id: Long): LiveData<Trash> = trashDao.getTrashById(id)
121  }
