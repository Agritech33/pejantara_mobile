26   import javax.inject.Singleton
27   
28   @Module
29   @InstallIn(SingletonComponent::class)
30   class AppModule {
31   
32       @Provides
33       @Singleton
34       fun provideApi(): ApiService {
35           return RetrofitFactory.makeRetrofitService()
36       }
37   
38       @Provides
39       @Singleton
40       fun provideDatabase(@ApplicationContext context: Context): Database {
41           return Room.databaseBuilder(
42               context, Database::class.java, "ecodo_db",
43           )
44               .fallbackToDestructiveMigration()
45               .build()
46       }
47   
48       @Provides
49       @Singleton
50       fun provideLaporanDao(database: Database): LaporanDao {
51           return database.laporanDao()
52       }
53   
54       @Provides
55       @Singleton
56       fun provideTrashDao(database: Database): TrashDao {
57           return database.trashDao()
58       }
59   
60       @Provides
61       @Singleton
62       fun provideTrashRepository(
63           apiService: ApiService,
64           database: Database,
65           tutorialRepository: TutorialRepository
66       ): TrashRepository {
67           return TrashRepository(
68               trashDao = database.trashDao(),
69               apiService = apiService,
70               tutorialRepository = tutorialRepository
71           )
72       }
73   
74       @Provides
75       @Singleton
76       fun provideVideoRepository(apiService: ApiService): TutorialRepository {
77           return TutorialRepository(apiService)
78       }
79   
80       @Provides
81       @Singleton
82       fun provideArticleRepository(apiService: ApiService): EdukasiRepository {
83           return EdukasiRepository(apiService)
84       }
85   
86       @Provides
87       @Singleton
88       fun provideFirebaseAuth(): FirebaseAuth {
89           return FirebaseAuth.getInstance()
90       }
91   
92       @Provides
93       @Singleton
94       fun provideGoogleSignInClient(@ApplicationContext context: Context): GoogleSignInClient {
95           val gso = GoogleSignInOptions.Builder(GoogleSignInOptions.DEFAULT_SIGN_IN)
96               .requestIdToken(context.getString(R.string.default_web_client_id)) // default_web_client_id dari google-services.json
97               .requestEmail()
98               .build()
99           return GoogleSignIn.getClient(context, gso)
100      }
101  
102      @Provides
103      @Singleton
104      fun provideFirebaseStorage(): FirebaseStorage {
105          return FirebaseStorage.getInstance()
106      }
107  
108      @Provides
109      @Singleton
110      fun provideFirebaseFirestore(): FirebaseFirestore {
111          return FirebaseFirestore.getInstance()
112      }
113  }
