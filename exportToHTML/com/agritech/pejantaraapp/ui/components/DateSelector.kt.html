1    package com.agritech.pejantaraapp.ui.components
2    
3    import android.app.DatePickerDialog
4    import android.content.Context
5    import android.widget.DatePicker
6    import androidx.compose.foundation.layout.*
7    import androidx.compose.material3.*
8    import androidx.compose.runtime.*
9    import androidx.compose.ui.Alignment
10   import androidx.compose.ui.Modifier
11   import androidx.compose.ui.unit.dp
12   import java.text.SimpleDateFormat
13   import java.util.*
14   
15   @Composable
16   fun DateSelector(
17       context: Context,
18       modifier: Modifier = Modifier,
19       onDateSelected: (String) -> Unit = {}
20   ) {
21       var selectedDate by remember { mutableStateOf("") }
22       val calendar = Calendar.getInstance()
23   
24       val datePickerDialog = DatePickerDialog(
25           context,
26           { _: DatePicker, year: Int, month: Int, dayOfMonth: Int ->
27               calendar.set(year, month, dayOfMonth)
28               val formatter = SimpleDateFormat("dd/MM/yyyy", Locale.getDefault()) // Compatible with API 25
29               selectedDate = formatter.format(calendar.time)
30               onDateSelected(selectedDate)
31           },
32           calendar.get(Calendar.YEAR),
33           calendar.get(Calendar.MONTH),
34           calendar.get(Calendar.DAY_OF_MONTH)
35       )
36   
37       Box(
38           modifier = modifier
39               .padding(16.dp)
40               .fillMaxWidth()
41       ) {
42           OutlinedButton(
43               onClick = { datePickerDialog.show() },
44               modifier = Modifier.align(Alignment.CenterStart)
45           ) {
46               Text(if (selectedDate.isEmpty()) "Pilih Tanggal" else selectedDate)
47           }
48       }
49   }
