44   @Composable
45   fun ScanScreen(
46       navController: NavController,
47       viewModel: ScanViewModel = hiltViewModel()
48   ) {
49       val context = LocalContext.current
50       val lifecycleOwner = LocalLifecycleOwner.current
51       val uploadState by viewModel.uploadState.collectAsState()
52   
53       // CameraX-related states
54       val cameraProviderFuture = remember { ProcessCameraProvider.getInstance(context) }
55       var preview by remember { mutableStateOf<Preview?>(null) }
56       var imageCapture by remember { mutableStateOf<ImageCapture?>(null) }
57       var cameraSelector by remember { mutableStateOf(CameraSelector.DEFAULT_BACK_CAMERA) }
58       val cameraExecutor: ExecutorService = remember { Executors.newSingleThreadExecutor() }
59   
60       // Handle Upload State
61       when (uploadState) {
62           is Resource.Loading -> {
63               // Tampilkan indikator loading jika diperlukan
64               Box(
65                   modifier = Modifier.fillMaxSize(),
66                   contentAlignment = Alignment.Center
67               ) {
68                   CircularProgressIndicator()
69               }
70           }
71           is Resource.Success -> {
72               Toast.makeText(context, "Deteksi Berhasil", Toast.LENGTH_SHORT).show()
73               val imagePath = (uploadState as Resource.Success<Long>).data.toString()
74               navController.navigate("result/$imagePath")
75               viewModel.resetUploadState()
76           }
77           is Resource.Error -> {
78               Toast.makeText(context, (uploadState as Resource.Error).error, Toast.LENGTH_SHORT).show()
79               viewModel.resetUploadState()
80           }
81       }
82   
83       // Gallery Launcher
84       val galleryLauncher = rememberLauncherForActivityResult(ActivityResultContracts.GetContent()) { uri ->
85           uri?.let { selectedUri ->
86               val file = uriToFile(selectedUri, context)
87               val reducedFile = file.reduceFileImage()
88               viewModel.uploadImage(reducedFile)
89           }
90       }
91   
92       // Camera Permission Launcher
93       val cameraPermissionLauncher = rememberLauncherForActivityResult(
94           ActivityResultContracts.RequestPermission()
95       ) { isGranted ->
96           if (!isGranted) {
97               Toast.makeText(context, "Camera permission denied", Toast.LENGTH_SHORT).show()
98           }
99       }
100  
101      // Request Camera Permission
102      LaunchedEffect(Unit) {
103          cameraPermissionLauncher.launch(android.Manifest.permission.CAMERA)
104      }
105  
106      // PreviewView setup
107      val previewView = remember { PreviewView(context) }
108  
109      // Initialize CameraX
110      LaunchedEffect(cameraProviderFuture) {
111          val cameraProvider = cameraProviderFuture.get()
112  
113          // Set up Preview Use Case
114          val newPreview = Preview.Builder().build().also { preview ->
115              preview.setSurfaceProvider(previewView.surfaceProvider)
116          }
117  
118          // Set up ImageCapture Use Case
119          val newImageCapture = ImageCapture.Builder().build()
120  
121          try {
122              cameraProvider.unbindAll()
123              cameraProvider.bindToLifecycle(
124                  lifecycleOwner,
125                  cameraSelector,
126                  newPreview,
127                  newImageCapture
128              )
129              preview = newPreview
130              imageCapture = newImageCapture
131          } catch (e: Exception) {
132              Log.e("ScanScreen", "Error binding camera: ${e.message}", e)
133          }
134      }
135  
136      // Main UI
137      Box(modifier = Modifier.fillMaxSize()) {
138          // Camera Preview
139          AndroidView(
140              factory = { previewView },
141              modifier = Modifier.fillMaxSize()
142          )
143  
144          // Controls Overlay
145          Box(
146              modifier = Modifier
147                  .fillMaxSize()
148                  .padding(16.dp)
149          ) {
150              Row(
151                  modifier = Modifier
152                      .fillMaxWidth()
153                      .align(Alignment.BottomCenter),
154                  horizontalArrangement = Arrangement.SpaceEvenly,
155                  verticalAlignment = Alignment.CenterVertically
156              ) {
157                  // Add from Gallery
158                  IconButton(
159                      onClick = { galleryLauncher.launch("image/*") },
160                  ) {
161                      Icon(
162                          painter = painterResource(id = R.drawable.ic_add_image),
163                          contentDescription = "Add Image",
164                          modifier = Modifier.size(120.dp)
165                      )
166                  }
167  
168                  // Capture Button
169                  Button(
170                      onClick = {
171                          lifecycleOwner.lifecycleScope.launch {
172                              val file = createCustomTempFile(context)
173                              val outputOptions = ImageCapture.OutputFileOptions.Builder(file).build()
174  
175                              imageCapture?.takePicture(
176                                  outputOptions,
177                                  cameraExecutor,
178                                  object : ImageCapture.OnImageSavedCallback {
179                                      override fun onImageSaved(output: ImageCapture.OutputFileResults) {
180                                          val reducedFile = file.reduceFileImage()
181                                          viewModel.uploadImage(reducedFile)
182                                      }
183  
184                                      override fun onError(exception: ImageCaptureException) {
185                                          Log.e("ScanScreen", "Photo capture failed", exception)
186                                      }
187                                  }
188                              )
189                          }
190                      },
191                      modifier = Modifier.size(80.dp),
192                  ) {
193                      Icon(
194                          painter = painterResource(id = R.drawable.ic_capture),
195                          contentDescription = "Capture"
196                      )
197                  }
198  
199                  // Switch Camera
200                  IconButton(
201                      onClick = {
202                          cameraSelector = if (cameraSelector == CameraSelector.DEFAULT_BACK_CAMERA) {
203                              CameraSelector.DEFAULT_FRONT_CAMERA
204                          } else {
205                              CameraSelector.DEFAULT_BACK_CAMERA
206                          }
207                      },
208                  ) {
209                      Icon(
210                          painter = painterResource(id = R.drawable.ic_switch_camera),
211                          contentDescription = "Switch Camera",
212                          modifier = Modifier.size(120.dp)
213                      )
214                  }
215              }
216          }
217      }
218  }
