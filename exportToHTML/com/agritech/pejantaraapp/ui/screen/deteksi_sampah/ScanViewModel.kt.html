15   import javax.inject.Inject
16   
17   @HiltViewModel
18   class ScanViewModel @Inject constructor(
19       private val trashRepository: TrashRepository
20   ) : ViewModel() {
21   
22       private val _uploadState = MutableStateFlow<Resource<Long>>(Resource.Loading)
23       val uploadState: StateFlow<Resource<Long>> = _uploadState.asStateFlow()
24   
25       fun uploadImage(imageFile: File) {
26           viewModelScope.launch {
27               _uploadState.value = Resource.Loading
28               trashRepository.predictTrash(imageFile).collect { resource ->
29                   _uploadState.value = resource
30               }
31           }
32       }
33   
34       fun resetUploadState() {
35           _uploadState.value = Resource.Loading
36       }
37   }
