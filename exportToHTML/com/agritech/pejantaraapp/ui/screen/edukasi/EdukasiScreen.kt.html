23   @OptIn(ExperimentalMaterial3Api::class)
24   @Composable
25   fun EdukasiScreen(viewModel: EdukasiViewModel) {
26       val articles by viewModel.articles.collectAsState()
27   
28       Log.d("EdukasiScreen", "Articles: $articles")
29   
30       Scaffold(
31           topBar = {
32               TopAppBar(
33                   title = { Text(text = "Edukasi", color = Color.White) },
34                   colors = TopAppBarDefaults.mediumTopAppBarColors(
35                       containerColor = Color(0xFF45624E)
36                   )
37               )
38           },
39           content = { padding ->
40               if (articles.isEmpty()) {
41                   Box(
42                       modifier = Modifier
43                           .fillMaxSize()
44                           .padding(padding),
45                       contentAlignment = Alignment.Center
46                   ) {
47                       CircularProgressIndicator()
48                   }
49               } else {
50                   LazyColumn(
51                       modifier = Modifier
52                           .fillMaxSize()
53                           .padding(padding)
54                   ) {
55                       items(articles) { article ->
56                           Log.d("EdukasiScreen", "Rendering Article: $article")
57                           EdukasiItemCard(article = article)
58                       }
59                   }
60               }
61           }
62       )
63   }
