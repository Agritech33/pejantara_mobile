1    package com.agritech.pejantaraapp.ui.screen.edukasi
2    
3    import android.util.Log
4    import androidx.lifecycle.ViewModel
5    import androidx.lifecycle.viewModelScope
6    import com.agritech.pejantaraapp.data.Article
7    import com.agritech.pejantaraapp.data.repository.EdukasiRepository
8    import kotlinx.coroutines.flow.MutableStateFlow
9    import kotlinx.coroutines.flow.StateFlow
10   import kotlinx.coroutines.launch
11   
12   class EdukasiViewModel(private val repository: EdukasiRepository) : ViewModel() {
13   
14       private val _articles = MutableStateFlow<List<Article>>(emptyList())
15       val articles: StateFlow<List<Article>> = _articles
16   
17       init {
18           loadArticles()
19       }
20   
21       private fun loadArticles() {
22           viewModelScope.launch {
23               val apiArticles = repository.fetchArticlesFromApi()
24               if (apiArticles != null) {
25                   Log.d("EdukasiViewModel", "Loaded Articles from API: $apiArticles")
26                   _articles.value = apiArticles
27               } else {
28                   val staticArticles = repository.fetchStaticArticles()
29                   Log.d("EdukasiViewModel", "Loaded Static Articles: $staticArticles")
30                   _articles.value = staticArticles
31               }
32           }
33       }
34   }
