47   @Composable
48   fun HistoryScanScreen(viewModel: HistoryScanViewModel) {
49       var selectedTab by remember { mutableStateOf("Riwayat Deteksi") }
50       val trashHistory by viewModel.trashHistory.observeAsState(emptyList())
51   
52       val validTabs = listOf("Riwayat Deteksi", "Riwayat Laporan")
53   
54       if (!validTabs.contains(selectedTab)) {
55           selectedTab = validTabs.first()
56       }
57   
58   
59       Column(
60           modifier = Modifier
61               .fillMaxSize()
62               .background(Color(0xFFF8F8F8))
63       ) {
64   
65           Box(
66               modifier = Modifier
67                   .fillMaxWidth()
68                   .height(88.dp)
69                   .background(Color(0xFF45624E)),
70               contentAlignment = Alignment.Center
71           ) {
72               Text(
73                   text = "Riwayat",
74                   color = Color.White,
75                   style = MaterialTheme.typography.titleLarge.copy(fontWeight = FontWeight.Bold)
76               )
77           }
78   
79           Spacer(modifier = Modifier.height(16.dp))
80   
81           TabRow(
82               selectedTab = selectedTab,
83               onTabSelected = { selectedTab = it }
84           )
85   
86           Spacer(modifier = Modifier.height(16.dp))
87   
88           if (trashHistory.isEmpty()) {
89               Box(
90                   modifier = Modifier
91                       .fillMaxSize()
92                       .padding(16.dp),
93                   contentAlignment = Alignment.Center
94               ) {
95                   Text(
96                       text = "Tidak ada riwayat yang ditemukan.",
97                       style = MaterialTheme.typography.bodyMedium,
98                       color = Color.Gray
99                   )
100              }
101          } else {
102              LazyColumn(
103                  contentPadding = PaddingValues(16.dp),
104                  verticalArrangement = Arrangement.spacedBy(16.dp),
105                  modifier = Modifier
106                      .fillMaxHeight()
107              ) {
108                  items(trashHistory) { trash ->
109                      HistoryScanCard(
110                          scan = ScanEntity(
111                              imageUrl = trash.imageData.toString(),
112                              title = trash.predictedClass,
113                              description = "Artikel: ${trash.articleUrl}"
114                          )
115                      )
116                  }
117              }
118          }
119      }
120  }

180  @Composable
181  fun TabRow(selectedTab: String, onTabSelected: (String) -> Unit) {
182      Row(
183          modifier = Modifier
184              .fillMaxWidth()
185              .padding(horizontal = 16.dp),
186          horizontalArrangement = Arrangement.SpaceBetween
187      ) {
188          TabItem(
189              title = "Riwayat Deteksi",
190              isSelected = selectedTab == "Riwayat Deteksi",
191              onClick = { onTabSelected("Riwayat Deteksi") },
192              modifier = Modifier.weight(1f)
193          )
194          TabItem(
195              title = "Riwayat Laporan",
196              isSelected = selectedTab == "Riwayat Laporan",
197              onClick = { onTabSelected("Riwayat Laporan") },
198              modifier = Modifier.weight(1f)
199          )
200      }
201  }
