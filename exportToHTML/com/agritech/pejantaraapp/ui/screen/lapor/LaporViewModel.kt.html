1    package com.agritech.pejantaraapp.ui.screen.lapor
2    
3    import androidx.compose.runtime.Composable
4    import androidx.lifecycle.ViewModel
5    import androidx.lifecycle.viewModelScope
6    import com.agritech.pejantaraapp.data.database.LaporanEntity
7    import com.agritech.pejantaraapp.data.repository.LaporanRepository
8    import dagger.hilt.android.lifecycle.HiltViewModel
9    import kotlinx.coroutines.launch
10   import javax.inject.Inject
11   
12   @HiltViewModel
13   class LaporViewModel @Inject constructor(
14       private val laporanRepository: LaporanRepository
15   ) : ViewModel() {
16   
17       fun tambahLaporan(
18           jenis: String,
19           deskripsi: String,
20           berat: Int,
21           tanggal: String,
22           onSuccess:  () -> Unit,
23           onFailure:  (String) -> Unit,
24           updateCoins: (Int) -> Unit
25       ) {
26           viewModelScope.launch {
27               try {
28                   val laporan = LaporanEntity(
29                       jenis = jenis,
30                       deskripsi = deskripsi,
31                       berat = berat,
32                       tanggal = tanggal
33                   )
34                   laporanRepository.insertLaporan(laporan)
35   
36                   val coinsEarned = berat * 1000
37                   updateCoins(coinsEarned)
38                   onSuccess()
39               } catch (e: Exception) {
40                   onFailure(e.localizedMessage ?: "Gagal menambahkan laporan")
41               }
42           }
43       }
44   }
