1    package com.agritech.pejantaraapp.ui.screen.login
2    
3    import android.app.Activity
4    import android.content.Context
5    import android.content.Intent
6    import androidx.lifecycle.ViewModel
7    import com.google.android.gms.auth.api.signin.GoogleSignInClient
8    import com.google.firebase.auth.FirebaseAuth
9    import com.google.firebase.auth.GoogleAuthProvider
10   import dagger.hilt.android.lifecycle.HiltViewModel
11   import kotlinx.coroutines.flow.MutableStateFlow
12   import kotlinx.coroutines.flow.StateFlow
13   import javax.inject.Inject
14   
15   @HiltViewModel
16   class LoginViewModel @Inject constructor(
17       private val auth: FirebaseAuth,
18       private val googleSignInClient: GoogleSignInClient
19   ) : ViewModel() {
20   
21       private val _isLoading = MutableStateFlow(false)
22       val isLoading: StateFlow<Boolean> = _isLoading
23   
24       fun loginWithEmailPassword(
25           email: String,
26           password: String,
27           onSuccess: () -> Unit,
28           onFailure: (String) -> Unit
29       ) {
30           if (email.isEmpty() || password.isEmpty()) {
31               onFailure("Email dan password tidak boleh kosong.")
32               return
33           }
34   
35           _isLoading.value = true
36           auth.signInWithEmailAndPassword(email, password)
37               .addOnCompleteListener { task ->
38                   _isLoading.value = false
39                   if (task.isSuccessful) {
40                       val user = auth.currentUser
41                       if (user?.isEmailVerified == true) {
42                           onSuccess()
43                       } else {
44                           onFailure("Email belum diverifikasi. Silahkan cek email Anda.")
45                       }
46                   } else {
47                       onFailure(task.exception?.message ?: "Error tidak diketahui")
48                   }
49               }
50       }
51   
52       fun loginWithGoogle(
53           idToken: String,
54           onSuccess: () -> Unit,
55           onFailure: (String) -> Unit
56       ) {
57           _isLoading.value = true
58           val credential = GoogleAuthProvider.getCredential(idToken, null)
59           auth.signInWithCredential(credential)
60               .addOnCompleteListener { task ->
61                   _isLoading.value = false
62                   if (task.isSuccessful) {
63                       onSuccess()
64                   } else {
65                       onFailure(task.exception?.message ?: "Google Sign-In gagal.")
66                   }
67               }
68       }
69   
70       fun getGoogleSignInIntent(): Intent {
71           return googleSignInClient.signInIntent
72       }
73   }
