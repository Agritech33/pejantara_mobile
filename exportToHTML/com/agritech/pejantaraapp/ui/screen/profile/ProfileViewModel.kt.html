37       private fun loadUserProfile() {
38           val userId = auth.currentUser?.uid ?: return
39           firestore.collection("users").document(userId).get()
40               .addOnSuccessListener { document ->
41                   val profile = document.toObject(ProfileState::class.java)
42                   profile?.let {
43                       _profileState.value = it
44                   }
45               }
46               .addOnFailureListener { e ->
47                   Log.e("ProfileViewModel", "Error memuat profil: ${e.message}")
48               }
49       }

99       fun uploadProfileImage(
100          context: Context,
101          uri: Uri,
102          onSuccess: () -> Unit,
103          onFailure: (String) -> Unit
104      ) {
105          val userId = auth.currentUser?.uid ?: return
106          val storageRef = firebaseStorage.reference.child("profile_images/$userId.jpg")
107  
108          val file = uriToFile(uri, context).reduceFileImage()
109          storageRef.putFile(Uri.fromFile(file))
110              .addOnSuccessListener {
111                  storageRef.downloadUrl.addOnSuccessListener { downloadUrl ->
112                      firestore.collection("users").document(userId)
113                          .update("photoUri", downloadUrl.toString())
114                          .addOnSuccessListener {
115                              _profileState.update { it.copy(photoUri = downloadUrl.toString()) }
116                              onSuccess()
117                          }
118                          .addOnFailureListener { e ->
119                              onFailure(e.message ?: "Gagal memperbarui URL gambar profil")
120                          }
121                  }
122              }
123              .addOnFailureListener { e ->
124                  onFailure(e.message ?: "Gagal mengunggah gambar")
125              }
126      }

128      fun updatePassword(
129          oldPassword: String,
130          newPassword: String,
131          onSuccess: () -> Unit,
132          onFailure: (String) -> Unit
133      ) {
134          val user = auth.currentUser
135          val email = user?.email
136  
137          if (user != null && !email.isNullOrEmpty()) {
138              val credential = EmailAuthProvider.getCredential(email, oldPassword)
139              user.reauthenticate(credential)
140                  .addOnCompleteListener { reauthTask ->
141                      if (reauthTask.isSuccessful) {
142                          user.updatePassword(newPassword)
143                              .addOnCompleteListener { updateTask ->
144                                  if (updateTask.isSuccessful) {
145                                      onSuccess()
146                                  } else {
147                                      onFailure(
148                                          updateTask.exception?.message ?: "Gagal mengubah password"
149                                      )
150                                  }
151                              }
152                      } else {
153                          onFailure(reauthTask.exception?.message ?: "Password lama salah")
154                      }
155                  }
156          } else {
157              onFailure("Pengguna tidak valid")
158          }
159      }

161      fun updateCoins(coins: Int) {
162          val userId = auth.currentUser?.uid ?: return
163          val currentCoins = _profileState.value.money ?: 0
164          val updatedCoins = currentCoins + coins
165  
166          firestore.collection("users").document(userId)
167              .update("money", updatedCoins)
168              .addOnSuccessListener {
169                  _profileState.update { it.copy(money = updatedCoins) }
170              }
171              .addOnFailureListener { e ->
172                  Log.e("ProfileViewModel", "Error memperbarui koin: ${e.message}")
173              }
174      }
