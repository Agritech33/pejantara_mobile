51   @OptIn(ExperimentalMaterial3Api::class)
52   @Composable
53   fun RegisterScreen(navController: NavController, viewModel: RegisterViewModel = hiltViewModel()) {
54       val nama = rememberSaveable { mutableStateOf("") }
55       val email = rememberSaveable { mutableStateOf("") }
56       val password = rememberSaveable { mutableStateOf("") }
57       val konfirmasiPassword = rememberSaveable { mutableStateOf("") }
58       val noHandphone = rememberSaveable { mutableStateOf("") }
59       val isLoading by viewModel.isLoading.collectAsState()
60       val context = LocalContext.current
61   
62       val googleSignInLauncher = rememberLauncherForActivityResult(
63           contract = ActivityResultContracts.StartActivityForResult()
64       ) { result ->
65           val task = GoogleSignIn.getSignedInAccountFromIntent(result.data)
66           try {
67               val account = task.getResult(ApiException::class.java)
68               account?.idToken?.let { idToken ->
69                   viewModel.registerWithGoogle(
70                       idToken = idToken,
71                       onSuccess = {
72                           Toast.makeText(context, "Login Google berhasil!", Toast.LENGTH_SHORT).show()
73                           navController.navigate("home")
74                       },
75                       onFailure = { error ->
76                           Toast.makeText(context, "Error: $error", Toast.LENGTH_SHORT).show()
77                       }
78                   )
79               }
80           } catch (e: ApiException) {
81               Toast.makeText(context, "Google Sign-In gagal: ${e.message}", Toast.LENGTH_SHORT).show()
82           }
83       }
84   
85       Column(
86           modifier = Modifier
87               .fillMaxSize()
88               .padding(16.dp),
89           verticalArrangement = Arrangement.Center,
90           horizontalAlignment = Alignment.Start
91       ) {
92           // Header
93           Text(text = "Daftar", fontSize = 28.sp, fontWeight = FontWeight.Bold)
94           Spacer(modifier = Modifier.padding(10.dp))
95           Text(text = "Segera daftarkan diri Anda ke aplikasi")
96   
97           // Input Fields
98           TextFieldWithLabel(value = nama.value, onValueChange = { nama.value = it }, label = "Nama Anda")
99           TextFieldWithLabel(value = email.value, onValueChange = { email.value = it }, label = "Email")
100          TextFieldWithLabel(
101              value = password.value,
102              onValueChange = { password.value = it },
103              label = "Password",
104              isPassword = true
105          )
106          TextFieldWithLabel(
107              value = konfirmasiPassword.value,
108              onValueChange = { konfirmasiPassword.value = it },
109              label = "Konfirmasi Password",
110              isPassword = true
111          )
112          TextFieldWithLabel(
113              value = noHandphone.value,
114              onValueChange = { noHandphone.value = it },
115              label = "No Handphone",
116              keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Number)
117          )
118  
119          // Button Daftar
120          Button(
121              onClick = {
122                  viewModel.onRegisterClick(
123                      email = email.value,
124                      password = password.value,
125                      confirmPassword = konfirmasiPassword.value,
126                      onSuccess = {
127                          Toast.makeText(context, "Registrasi berhasil!", Toast.LENGTH_SHORT).show()
128                          navController.navigate("verifikasi_email/${email.value}")
129                      },
130                      onFailure = { error ->
131                          Toast.makeText(context, error, Toast.LENGTH_SHORT).show()
132                      }
133                  )
134              },
135              modifier = Modifier.fillMaxWidth(),
136              enabled = !isLoading
137          ) {
138              if (isLoading) {
139                  CircularProgressIndicator(color = Color.White, strokeWidth = 2.dp)
140              } else {
141                  Text(text = "Daftar", color = Color.White)
142              }
143          }
144  
145          // Opsi Login dengan Google
146          Button(
147              onClick = {
148                  val gso = GoogleSignInOptions.Builder(GoogleSignInOptions.DEFAULT_SIGN_IN)
149                      .requestIdToken(context.getString(R.string.default_web_client_id))
150                      .requestEmail()
151                      .build()
152                  val googleSignInClient = GoogleSignIn.getClient(context, gso)
153                  googleSignInLauncher.launch(googleSignInClient.signInIntent)
154              },
155              modifier = Modifier.fillMaxWidth(),
156              colors = ButtonDefaults.buttonColors(containerColor = Color.Red)
157          ) {
158              Text(text = "Sign in dengan Google", color = Color.White)
159          }
160      }
161  }

164  @OptIn(ExperimentalMaterial3Api::class)
165  @Composable
166  fun TextFieldWithLabel(
167      value: String,
168      onValueChange: (String) -> Unit,
169      label: String,
170      isPassword: Boolean = false,
171      keyboardOptions: KeyboardOptions = KeyboardOptions.Default
172  ) {
173      Text(text = label)
174      Spacer(modifier = Modifier.padding(5.dp))
175      TextField(
176          value = value,
177          onValueChange = onValueChange,
178          modifier = Modifier.fillMaxWidth(),
179          colors = TextFieldDefaults.textFieldColors(containerColor = Color(0xFFE5E5E5)),
180          visualTransformation = if (isPassword) PasswordVisualTransformation() else VisualTransformation.None,
181          keyboardOptions = keyboardOptions
182      )
183  }
