15   import java.security.SecureRandom
16   import javax.inject.Inject
17   
18   @HiltViewModel
19   class RegisterViewModel @Inject constructor(
20       private val auth: FirebaseAuth,
21       private val googleSignInClient: GoogleSignInClient
22   ) : ViewModel() {
23   
24       private val _isLoading = MutableStateFlow(false)
25       val isLoading: StateFlow<Boolean> = _isLoading
26   
27       private var verificationCode: String = generateRandomCode()
28   
29       fun onRegisterClick(
30           email: String,
31           password: String,
32           confirmPassword: String,
33           onSuccess: () -> Unit,
34           onFailure: (String) -> Unit
35       ) {
36           when {
37               email.isEmpty() || password.isEmpty() || confirmPassword.isEmpty() -> {
38                   onFailure("Semua field harus diisi.")
39               }
40               !android.util.Patterns.EMAIL_ADDRESS.matcher(email).matches() -> {
41                   onFailure("Format email tidak valid.")
42               }
43               password.length < 6 -> {
44                   onFailure("Password harus memiliki minimal 6 karakter.")
45               }
46               password != confirmPassword -> {
47                   onFailure("Password dan konfirmasi password tidak cocok.")
48               }
49               else -> {
50                   registerWithEmailPassword(email, password, onSuccess, onFailure)
51               }
52           }
53       }
54   
55       private fun registerWithEmailPassword(
56           email: String,
57           password: String,
58           onSuccess: () -> Unit,
59           onFailure: (String) -> Unit
60       ) {
61           _isLoading.value = true
62           auth.createUserWithEmailAndPassword(email, password)
63               .addOnCompleteListener { task ->
64                   _isLoading.value = false
65                   if (task.isSuccessful) {
66                       val user = auth.currentUser
67                       user?.let {
68                           sendEmailVerification(user, onSuccess, onFailure)
69                       } ?: onFailure("User tidak ditemukan setelah registrasi.")
70                   } else {
71                       onFailure(task.exception?.message ?: "Terjadi kesalahan.")
72                   }
73               }
74       }
75   
76       private fun sendEmailVerification(
77           user: FirebaseUser,
78           onSuccess: () -> Unit,
79           onFailure: (String) -> Unit
80       ) {
81           user.sendEmailVerification()
82               .addOnCompleteListener { task ->
83                   if (task.isSuccessful) {
84                       onSuccess()
85                   } else {
86                       onFailure(task.exception?.message ?: "Gagal mengirim email verifikasi.")
87                   }
88               }
89       }
90   
91       fun registerWithGoogle(
92           idToken: String,
93           onSuccess: () -> Unit,
94           onFailure: (String) -> Unit
95       ) {
96           _isLoading.value = true
97           val credential = GoogleAuthProvider.getCredential(idToken, null)
98           auth.signInWithCredential(credential)
99               .addOnCompleteListener { task ->
100                  _isLoading.value = false
101                  if (task.isSuccessful) {
102                      onSuccess()
103                  } else {
104                      onFailure(task.exception?.message ?: "Google Sign-In gagal.")
105                  }
106              }
107      }
108  
109      fun resendVerificationEmail(
110          onSuccess: () -> Unit,
111          onFailure: (String) -> Unit
112      ) {
113          val user = auth.currentUser
114          user?.let {
115              user.sendEmailVerification()
116                  .addOnCompleteListener { task ->
117                      if (task.isSuccessful) {
118                          onSuccess()
119                      } else {
120                          onFailure(task.exception?.message ?: "Gagal mengirim ulang email verifikasi.")
121                      }
122                  }
123          } ?: onFailure("User tidak ditemukan.")
124      }
125  
126      fun verifyEmailCode(
127          enteredCode: String,
128          onSuccess: () -> Unit,
129          onFailure: (String) -> Unit
130      ) {
131          if (enteredCode == verificationCode) {
132              onSuccess()
133          } else {
134              onFailure("Kode verifikasi salah. Silakan coba lagi.")
135          }
136      }
137  
138      private fun generateRandomCode(): String {
139          val random = SecureRandom()
140          return (random.nextInt(9000) + 1000).toString() // Generate 4-digit code
141      }
142  }
