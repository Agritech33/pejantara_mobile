27   fun uriToFile(imageUri: Uri, context: Context): File {
28       val tempFile = createCustomTempFile(context)
29       val inputStream = context.contentResolver.openInputStream(imageUri) as InputStream
30       val outputStream = FileOutputStream(tempFile)
31       val buffer = ByteArray(1024)
32       var length: Int
33   
34       while (inputStream.read(buffer).also { length = it } > 0) {
35           outputStream.write(buffer, 0, length)
36       }
37   
38       outputStream.close()
39       inputStream.close()
40       return tempFile
41   }
42   fun createCustomTempFile(context: Context): File {
43       val filesDir = context.externalCacheDir
44       val timeStamp = SimpleDateFormat(FILENAME_FORMAT, Locale.US).format(Date())
45       return File.createTempFile(timeStamp, ".jpg", filesDir)
46   }

48   fun File.reduceFileImage(): File {
49       val bitmap = BitmapFactory.decodeFile(this.path).getRotatedBitmap(this)
50       var compressQuality = 100
51       var streamLength: Int
52   
53       do {
54           val bmpStream = ByteArrayOutputStream()
55           bitmap?.compress(Bitmap.CompressFormat.JPEG, compressQuality, bmpStream)
56           val bmpPicByteArray = bmpStream.toByteArray()
57           streamLength = bmpPicByteArray.size
58           compressQuality -= 5
59       } while (streamLength > MAXIMAL_SIZE)
60   
61       bitmap?.compress(Bitmap.CompressFormat.JPEG, compressQuality, FileOutputStream(this))
62       return this
63   }

65   fun Bitmap.getRotatedBitmap(file: File): Bitmap? {
66       val orientation = ExifInterface(file).getAttributeInt(
67           ExifInterface.TAG_ORIENTATION, ExifInterface.ORIENTATION_UNDEFINED
68       )
69   
70       return when (orientation) {
71           ExifInterface.ORIENTATION_ROTATE_90 -> rotateImage(this, 90F)
72           ExifInterface.ORIENTATION_ROTATE_180 -> rotateImage(this, 180F)
73           ExifInterface.ORIENTATION_ROTATE_270 -> rotateImage(this, 270F)
74           else -> this
75       }
76   }

78   fun rotateImage(source: Bitmap, angle: Float): Bitmap {
79       val matrix = Matrix()
80       matrix.postRotate(angle)
81       return Bitmap.createBitmap(
82           source, 0, 0, source.width, source.height, matrix, true
83       )
84   }
